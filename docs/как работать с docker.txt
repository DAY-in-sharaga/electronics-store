# какие контейнеры работают прямо сейчас
docker ps  

# Она покажет список запущенных контейнеров, включая:
# CONTAINER ID — уникальный ID контейнера
# IMAGE — образ, из которого он запущен
# COMMAND — команда, которую выполняет
# STATUS — сколько времени работает
# PORTS — проброшенные порты
# NAMES — имя контейнера

# Останавливает контейнер (7d4986e329b5 - CONTAINER ID)
docker stop 7d4986e329b5

# Удаляет контейнер
docker rm 7d4986e329b5
